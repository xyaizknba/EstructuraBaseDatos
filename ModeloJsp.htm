public java.util.List getDatosPersona(String sSQL) throws java.sql.SQLException {
  java.util.List listaPersona = new ArrayList();
  java.sql.Connection dbConnection = getConnection();
  java.sql.Statement dbStatement = dbConnection.createStatement();
  java.sql.ResultSet resultados = dbStatement.executeQuery(sSQL);
  while(resultados.next()){
   Persona oPersona = new Persona();

   oPersona.setNombre(resultados.getString("nombre"));
   oPersona.setClave(resultados.getString("clave"));
   oPersona.setCorreo(resultados.getString("correo"));
   oPersona.setNumero(resultados.getString("numero"));
   oPersona.setTelefono(resultados.getString("telefono"));
   oPersona.setComentario(resultados.getString("comentario"));

   oPersona.setFecha(resultados.getString("fecha"));
   oPersona.setFechapago(resultados.getString("fecha_pago"));
   oPersona.setColor(resultados.getString("color"));
   oPersona.setPrecio(resultados.getString("precio"));
   oPersona.setMoneda(resultados.getString("moneda"));
   oPersona.setIddeporte(resultados.getString("id_deporte"));
