Select Fechas Con el siguiente Formato Dia-Mes-Año 08-08-1975

SELECT persona.id_persona, persona.nombre, persona.clave, persona.correo, persona.numero, persona.telefono, persona.comentario, persona.piezas, persona.soltero, persona.auto, persona.id_ciudad, IFNULL(DATE_FORMAT(persona.fecha,CONCAT('%d','-','%m','-','%Y')),'') AS DteFmt_fecha, IFNULL(DATE_FORMAT(persona.fecha_pago,CONCAT('%d','-','%m','-','%Y')),'') AS DteFmt_fecha_pago, persona.color, persona.precio, persona.moneda, persona.id_deporte, persona.fecha_contratacion, persona.id_empresa, persona.fechainsercion, persona.fechaactualizacion, persona.fechaeliminacion, persona.registrovisible FROM persona

Seleccionar 1 Registro
